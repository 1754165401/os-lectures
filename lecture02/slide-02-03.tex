  }

}

\begin{frame}[fragile]
\frametitle{高级语言编译到机器指令}
\framesubtitle{汇编代码}
	
\begin{lstlisting}
# sum_to(n)
# expects argument in a0
# returns result in a0
sum_to:
mv t0, a0          # t0 <- a0
li a0, 0           # a0 <- 0
loop:
add a0, a0, t0     # a0 <- a0 + t0
addi t0, t0, -1    # t0 <- t0 - 1
bnez t0, loop      # if t0 != 0: pc <- loop
ret
\end{lstlisting}	

\begin{itemize}
	
	\item 有限的抽象
	\begin{itemize}
	  \item 无类型的位置参数 -- 没有局部变量 -- 仅寄存器
	\end{itemize}
		
\end{itemize}
	
\end{frame}

	\item RUST代码遵循此约定
